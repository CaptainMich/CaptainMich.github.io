Dictionary

D ictionary is a collection which is unordered, changeable and indexed 


# -------------------------------------------------------------------------------------------------
# DICTIONARY
# -------------------------------------------------------------------------------------------------

profile = {                                       # define a dictionary
  1: "Captain",                                   # (key : value)
  2: "Mich",                                      # ...
  "Age": 100,                                     # ...
  "Job": "Programmer"                             # ...
}                                                 # ...

print(profile)                                    # ...

  

# -------------------------------------------------------------------------------------------------
# WORKING with DICTIONARY
# -------------------------------------------------------------------------------------------------

print(profile.items())                            # access all items (1)
print('')                                         # ...

for k,v in profile.items():                       # access all items (2)
  print(str(k) + ":" + str(v))                    # ...
print('')

keys = profile.keys()                             # access keys
for k in keys:                                    # and iterate through them
  print(k)                                        # ...
print('')

values = profile.values()                         # access values
for v in values:                                  # and iterate through them
  print(v)                                        # ...
print('')

print(profile["Job"])                             # access value by passing the key

del profile[2]                                    # remove element by passing the key
print(profile)                                    # ...
