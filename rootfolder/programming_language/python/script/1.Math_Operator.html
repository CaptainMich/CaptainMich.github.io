Math Operator

B elow we cover all the default math operator and how to work with them 


# --------------------------------------------------------------------------------------
# MATH OPERATOR
# --------------------------------------------------------------------------------------

print('\n\t\tMATH OPERATOR\n')

print(3 + 4)                        # example of add
print(4 - 3)                        # example of subtraction
print(6 * 5)                        # example of multiplication
print(12 / 5)                       # example of division
print(18 // 4)                      # example of division(without rest)
print(8 % 5)                        # example of reminder
print(5 ** 3)                       # example of power of number

print(8 + 2 * 10)                   # example of order of operations (1. not ordered)
print(8 + (2 * 10))                 # example of order of operations (2. ordered)

print(float('8.5') + 5)             # example of conversion(2)

Exercise 3.
Write a Python program that compute the greatest common divisor (GCD) and the least common multiple (LCM) of two positive integers. 

# -------------------------------------------------------------------------------------------------
# Exercise_03
# Write a Python program that compute the greatest common divisor (GCD)  and the least common
#  multiple (LCM) of two positive integers.
# -------------------------------------------------------------------------------------------------

def gcd(x, y):
  while y != 0:
    (x, y) = (y, x % y)
  return abs(x)

def lcm(x,y):
  return ((x * y)/ gcd(x,y))

print('\nExercise_03')

firstNumber = int(input('Please enter first number: '))
secondNumber = int(input('Please enter second number: '))

print('GCD between ' + str(firstNumber) + ' and ' + str(secondNumber)
        + ' is ' + str(gcd(firstNumber,secondNumber)))
print('LCM between ' + str(firstNumber) + ' and ' + str(secondNumber)
        + ' is ' + str(lcm(firstNumber,secondNumber)) + '\n')

Exercise 6.
Write a Python program that calculate the Fibonacci series (not recursive way). 

# -------------------------------------------------------------------------------------------------
# Exercise_06
# Write a Python program that calculate the Fibonacci series (not recursive way).
# -------------------------------------------------------------------------------------------------


def Fibonacci():
  a, b = 0, 1
  while True:                                                     # First iteration:
      yield a                                                     # yield 0 to start with and then
      a, b = b, a + b                                             # a will now be 1, and b will also be 1, (0 + 1)

print('\nExercise_06')
index_number = int(input('Please enter the number of terms: '))

for index, fibonacci_number in enumerate(Fibonacci()):
  print('{i:3}: {f:3}'.format(i=index, f=fibonacci_number))
  if index == index_number:
      break
