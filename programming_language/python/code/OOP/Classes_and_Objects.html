Classes and Objects

B elow we cover what a class is and how to work with it.


Classes

Classes allow us to create our own data types that is user defined data types. So, a class is like an object constructor, or a "blueprint" for creating objects. Please note that almost everything in Python is an object, with its properties and methods

    self.name = "IPhone"                          # note that self points to the
    self.description = "Made by Apple"            # current object that is being

# -------------------------------------------------------------------------------------------------
# INSTANCE METHOD
# -------------------------------------------------------------------------------------------------

class Course:

    def __init__(self, name, ratings):                # ...
      self.name = name                                # ...
      self.ratings = ratings                          # ...

    def average(self):                                # defining a method
      number_of_ratings = len(self.ratings)           # ...
      average = sum(self.ratings)/ number_of_ratings  # ...
      print("Average Ratings for {} course "
            "is {}".format(self.name, average))       # ...

    def display(self):                                #
      print(self.name)                                # display class field
      print(self.ratings)                             # ...

course_py = Course("Python", [1,2,3,4,5])             # initialize two new
course_java = Course("Java", [4,5,6,7])               # course class object

course_py.display()                                   # access class method
course_py.average()                                   # ...

print('')

course_java.display()                                 # access class method
course_java.average()                                 # ...
  


# -------------------------------------------------------------------------------------------------
# GETTER and SETTER
# -------------------------------------------------------------------------------------------------

class Programmer:

  def setName(self, name):              # define a method to set variable (1)
    self.name = name                    # assign the variable

  def getName(self):                    # define a method to get variable (1)
    return self.name                    # get the variable

  def setSalary(self, sal):             # define a method to set variable (2)
    self.salary = sal                   # assign the variable

  def getSalary(self):                  # define a method to set variable (2)
    return self.salary                  # get the variable

p = Programmer()                        # ...
p.setName("Captain")                    # ...
p.setSalary(200000)                     # ...
print(p.getName())                      # ...
print(p.getSalary())                    # ...

class Student:

# -------------------------------------------------------------------------------------------------
# STATIC METHOD
# -------------------------------------------------------------------------------------------------

class ObjectCounter:

  number_of_obj = 0                                           # create a static variable

  def __init__(self):                                         # ...
    print("Object Created")                                   # ...
    ObjectCounter.number_of_obj += 1                          # ...

  @staticmethod                                               # define a static method
  def displayCount():                                         # ...
    print("Number of Objects:", ObjectCounter.number_of_obj)  # ...

o1 = ObjectCounter()                                          # create an object
o2 = ObjectCounter()                                          # ...
o3 = ObjectCounter()                                          # ...
o4 = ObjectCounter()                                          # ...
ObjectCounter.displayCount()                                  # show number of created obj
  


# -------------------------------------------------------------------------------------------------
# INNER CLASS
# -------------------------------------------------------------------------------------------------

class Car:                                  # ...
  def __init__(self, model, cv):            # ...
    self.model = model                      # ...
    self.cv = cv                            # ...

  class Engine:                             # define an inner class
    def __init__(self, id):                 # ...
      self.id = id                          # ...

    def start(self):                        # ...
      print("Engine started")               # ...

car = Car("Ferrari", 1000)                  # ...
engine = car.Engine(6754)                   # create inner class object
engine.start()                              # access inner class method
