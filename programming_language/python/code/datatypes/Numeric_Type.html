Numeric Types

B elow we cover all the default Numeric Types and how to work with them 


# --------------------------------------------------------------------------------------
# NUMERIC TYPES
# --------------------------------------------------------------------------------------

# Integers
a = 5                               # example of integer (1)
print(a)                            # ...
print(type(a))                      # <class 'int'>
print('')                           # N.B: in python everything is an object

# Floating Points
x = 33.5                            # example of floating point (2)
print(x)                            # ...
print(type(x))                      # <class 'float'>
print('')                           # ...

# Complex Types
c = 4 + 5j                          # example of complex (3)
print(c)                            # ...
print(type(c))                      # <class 'complex'>
print('')                           # ...

# Binary/Hexadecimal Values
b = 0b01010                         # example of binary integer (4)
h = 0xf                             # example of hexadecimal integer (5)
print(b)                            # ...
print(h)                            # ...
print(type(b))                      # <class 'int'>
print(type(h))                      # <class 'float'>
print('')                           # ...

# Boolean Types
g = True                            # example of boolean type (6)
print(g)                            # ...
print(type(g))                      # <class 'boolean'>
print(9>8)                          # condition true
print(9<8)                          # condition false
print('')                           # ...

  

# --------------------------------------------------------------------------------------
# Math Operator
# --------------------------------------------------------------------------------------

print(int('42'))                      # example of conversion(1)
print(float('42'))                    # example of conversion(2)
print(bin(42))                        # example of conversion(3)
print(hex(42))                        # example of conversion(4)
print(oct(42))                        # example of conversion(5)

print("Addition:\t", 3 + 4)           # example of add
print("Subtraction:\t", 4 - 3)        # example of subtraction
print("Mul:\t\t", 6 * 5)              # example of multiplication
print("Div:\t\t", 12 / 5)             # example of division
print("Floor Div:\t", 18 // 4)        # example of division(without rest)
print("Mod:\t\t", 8 % 5)              # example of reminder
print("Pow:\t\t", 5 ** 3)             # example of power of number

print(8 + 2 * 10)                     # example of order of operations (1. not ordered)
print(8 + (2 * 10))                   # example of order of operations (2. ordered)

a, b = 10,5

a += b                                # assignment operator (1)
print(a)                              # ...
a *= b                                # assignment operator (2)
print(a)                              # ...

print(a==75 and b==5)                 # logical operator (1)
print(a==10 and b==5)                 # logical operator (2)
print(a==10 and b==5)                 # logical operator (3)
print(not(a==10 and b==5))            # logical operator (4)

Exercise 3.
Write a Python program that compute the greatest common divisor (GCD) and the least common multiple (LCM) of two positive integers. 

# -------------------------------------------------------------------------------------------------
# Exercise_03
# Write a Python program that compute the greatest common divisor (GCD)  and the least common
#  multiple (LCM) of two positive integers.
# -------------------------------------------------------------------------------------------------

def gcd(x, y):
  while y != 0:
    (x, y) = (y, x % y)
  return abs(x)

def lcm(x,y):
  return ((x * y)/ gcd(x,y))

firstNumber = int(input('Please enter first number: '))
secondNumber = int(input('Please enter second number: '))

print('GCD between ' + str(firstNumber) + ' and ' + str(secondNumber)
        + ' is ' + str(gcd(firstNumber,secondNumber)))
print('LCM between ' + str(firstNumber) + ' and ' + str(secondNumber)
        + ' is ' + str(lcm(firstNumber,secondNumber)) + '\n')

# -------------------------------------------------------------------------------------------------

# Write a Python program that calculate the Fibonacci series (not recursive way).
# -------------------------------------------------------------------------------------------------

def Fibonacci():
  a, b = 0, 1
  while True:                       # First iteration:
      yield a                       # yield 0 to start with and then
      a, b = b, a + b               # "a" will now be 1, and "b" will also be 1, (0 + 1)

index_number = int(input('Please enter the number of terms: '))

for index, fibonacci_number in enumerate(Fibonacci()):
  print('{i:3}: {f:3}'.format(i=index, f=fibonacci_number))
  if index == index_number:
      break
