List

L ist is a collection which is ordered and changeable. Allows duplicate members 


# -------------------------------------------------------------------------------------------------
# LISTS
# -------------------------------------------------------------------------------------------------

empty = []                                      # empty list
print(empty)                                    # ...
players = [70,5,9]                              # syntax of a list
print(players)                                  # print entire list

different = [10,20, "Newton", 30.5, True]       # define a different element type list
print(different)                                # ...



# -------------------------------------------------------------------------------------------------
# WORK with LIST
# -------------------------------------------------------------------------------------------------

print(len(different))                 # to see the lenght of the list
print(different[2])                   # to access a member of a list in the position ['x']
different[1] = 90                     # to permanently change the value of a member
print(different)                      # ...

print(different + [0,1,2])            # concatenate list to another list
print(different)                      # it is not a permanently change

players = players * 2                 # multiply the list
print(players)                        # ...

players.append(4)                     # to permanently add an element to the list
print(players)                        # ...

players.insert(6, 99)                 # to permanently add an (element y, at index x)
print(players)                        # ...

players.remove(70)                    # to permanently remove an element to the list
print(players)                        # ...

players.pop(2)                        # to permanently remove an element to the list
print(players)                        # (position of the list is indicate in brackets)

del(players[3])                       # to permanently remove an element at index [x]
print(players)                        # ...

players.sort()                        # to order list in ascending fashion
print(players)                        # ...

players.sort(reverse=True)            # to reverse list (1)
print(players)                        # ...

players.reverse()                     # to reverse list (2)
print(players)                        # ...

print(max(players))                   # to get the maximum value of the list
print(min(players))                   # to get the minimum value of the list

                                        # uncomment this line to see the error

print('')

print(different[0:4])                 # to slicing up a list [from element 'x' to(:)
print(different[2:3])                 # element 'y']
print(different[2:])                  # ...
print(different[:])                   # ...

print('')

different[:3] = [0,1,2]               # to change multiple values [from element 'x'
print(different)                      # to(:) element 'y']
different[:2] = [3,4,5]               # if y is less than the number of replacing values
print(different)                      # element will be added to the list

different[:2] = []                    # to delete items from the list(1)
print(different)                      # ...

different[:] = []                     # to delete the entire list
print(different)                      # ...

players.clear()                       # to remove all item from the list
print(players)                        # ...




# -------------------------------------------------------------------------------------------------
# NESTING and List COMPREHENSION
# -------------------------------------------------------------------------------------------------

matrix = [[1,2,3],                                      # 3 x 3 Matrix, as nested list
          [4,5,6],                                      # ...
          [7,8,9]]                                      # ...

print(matrix)                                           # print entire list
print(matrix[1])                                        # print row 2
print(matrix[1][2], '\n')                               # print element 3 of the row 2

col2 = [row[1] for row in matrix]                       # collect the items in column 2
print(col2)                                             # ...

print(matrix)                                           # matrix is unchanged
print('')
                                                        # .. some other examples
print([row[1] + 1 for row in matrix])                   # add 1 to each item in column 2
print([row[1] for row in matrix if row[1] % 2 == 0])    # filter out odd items
print('')                                               # of column 2

diag = [matrix[i][i] for i in range(0, len(matrix))]    # collect diagonal from matrix
print(diag)                                             # to see the result

doubles = [letter * 2 for letter in 'spam']             # repeat character in a string
print(doubles, '\n')                                    # to see the result

Generator = (sum(row) for row in matrix)                # create a generator of row sum
print(next(Generator))                                  # (sum items in a sequence)
print(next(Generator))                                  # run the iteration protocol
print(next(Generator))                                  # next()


print({sum(row) for row in matrix})                     # create a set of row sums
print({i : sum(matrix[i]) for i in range(3)})           # creates key/value table of
print('')                                               # row sums

print([ord(x) for x in 'spaam'])                        # list of character ordinals
print({ord(x) for x in 'spaam'})                        # Set remove duplicates
print({x: ord(x) for x in 'spaam'})                     # dictionary keys are unique
