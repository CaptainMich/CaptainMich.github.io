# -------------------------------------------------------------------------------------------------
# WORK with STRING
# -------------------------------------------------------------------------------------------------

user = 'Tuna is the best'

print(user[0])                          # use it if you want a specif letter of a string
print(user[1])                          # ...
print(user[2])                          # ...
print(user[3])                          # ...
print('')

print(user[-1])                         # ...if you want to start from the end
print(user[-2])                         # ...
print(user[-3])                         # ...
print(user[-4])                         # ...
print('')

print(user[1:3])                        # to slicing up a string[from character 'x' to(:)
print(user[:3])                         # character 'y' not including 'y')
print(user[1:])                         # ...
print(user[:])                          # ...
print(user[-3:-1])                      # ...
print(user[0:4:3])                      # define also a step
print(user[::-1])                       # to reverse the string
print('')

print(len(user))                        # to measure the lenght of a string
print(len('Tuna'))                      # (N.B: blank space count as character)
print(user.count('e'))                 # count the number of occurences of substring

print(user.find('un'))                  # find the offset of a substring in 'user';
                                        # return the index if the substring is found
print(user.find('u',0,len(user)))       # also define delimiters
print(user.replace('una', 'omb'))       # replace the 'una' occurence of user with 'omb'

print(user)                             # notice that the originally string is not
                                        # permanently modified

print(user.upper())                     # convert all the contenent upper
print(user.lower())                     # convert all the contenent lowercase
print(user.title())                     # convert the contenent into titlecased version

print(user.isalpha())                   # find out if all the character in the string are
                                        # alphabetic and return true if there is at least
                                        # one character, false otherwise

line = '  aaa,bbb,cccccc,dd  \n'        # ...

print(line.split(','))                  # split on a specific delimiter into a list of
                                        # substring
print(line.strip())                     # remove all whitespaces
print(line.rstrip())                    # remove whitespace characters on the right side
print(line.lstrip().split())            # combine two operation


print('%s, eggs, and %s' % ('spam', 'SPAM!'))           # formatting expression
print('{}, eggs, and {}'.format('spam', 'SPAM!'))       # formatting method




# -------------------------------------------------------------------------------------------------
# PATTERN MATCHING
# -------------------------------------------------------------------------------------------------

import re

str_example = 'Hello    Python world'                      # search for a substring that begins with the word
match = re.match('Hello[\t]*(.*)world', str_example)       # "Hello" followed by zero or more tabs or space,
print(match.group(1))                                      # followed by arbitrary characters saved as matched
                                                           # group (avaiable only if a substring is found)

pattern = '/usr/home/testuser'                             # another example that picks out three groups
match = re.match('[/:](.*)[/:](.*)[/:](.*)', pattern)      # separated by slashes
print(match.groups())                                      # ...

match = re.split('[/:]', pattern)                          # in this case split give out the same result as
print(match)                                               # previous example
