Set

S et is a collection which is unordered and unindexed. No duplicate members are allowed 


# -------------------------------------------------------------------------------------------------
# SET
# -------------------------------------------------------------------------------------------------

groceries = {'cereal', 'milk', 'starcrunch', 'beer',    # set are list that can't
              'duct tape', 'lotion', 'beer'}            # containe any duplicate ...
print(groceries)                                        # beer will be print only once

if 'milk' in groceries:                                 # you can easily check if an
    print('You already have milk hoss!\n')              # element is inside the set (1)
else:                                                   # ...
    print('You need milk!\n')                           # ...

  

# -------------------------------------------------------------------------------------------------
# WORKING with SET
# -------------------------------------------------------------------------------------------------

groceries.update(['cheese', 'watermelon'])              # add element to set
print(groceries)                                        # ...
groceries.remove('cereal')                              # remove element to set
print(groceries)                                        # ...
