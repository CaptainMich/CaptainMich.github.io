Tuple

T uple is a collection which is ordered and unchangeable. Allows duplicate members. 


# -------------------------------------------------------------------------------------------------
# TUPLE
# -------------------------------------------------------------------------------------------------

tpl = ()                                          # define an empty tuple
print(tpl)                                        # ...

one_element = (24, )                              # define a single element tuple
print(one_element)                                # ...

numbers = (1,2,3,4,1)                             # define 5-items tuple
print(numbers)                                    # ...

mixed_tuple = 'spam', 3.0, [10,20,30]             # pharenteses can be omitted
print(mixed_tuple)                                # ...



# -------------------------------------------------------------------------------------------------
# WORK with Tuple
# -------------------------------------------------------------------------------------------------

print(numbers * 2)                                # repetition
print(numbers + (5,6))                            # concatenation
print(numbers)                                    # ...

print(numbers[0])                                 # indexing (1)
print(mixed_tuple[2][1])                          # indexing (2)
print(numbers.index(4))                           # type-specific method:
                                                  # number four appears at offset 3

print(numbers.count(1))                           # counts how many times the
                                                  # argument occured

                                                  # uncomment to see the error
#numbers[0] = 2                                   # Tuple are immutable(1)
#mixed_tuple.append(234)                          # Tuple are immutable(2)
mutable_numbers = (2,) + numbers[1:]              # create a new tuple for change
print(mutable_numbers)                            # value


lst = ['Einstein', 23, 98.5]                      # converting list into a tuple
converted = tuple(lst)                            # ...
print(converted)                                  # ...
print(type(converted))                            # ...
